#!/bin/bash

git_path="/opt/homebrew/bin/git"

# 1. ввод ссылки
repo_url=$(osascript -e 'text returned of (display dialog "Имя месту содержания Дитя" default answer "")')

if [ -z "$repo_url" ]; then
    osascript -e 'display alert "Ссылка не указана. Выход."'
    exit 1
fi

# 2. выбор сложения
folder_path=$(osascript -e 'POSIX path of (choose folder with prompt "Выбери сложение для размещения:")')

if [ -z "$folder_path" ]; then
    osascript -e 'display alert "Сложение не выбрано. Выход."'
    exit 1
fi

cd "$folder_path" || {
    osascript -e 'display alert "Не удалось перейти в сложение"'
    exit 1
}

# 3. принятие
$git_path clone "$repo_url"

if [ $? -eq 0 ]; then
    osascript -e 'display notification "Сложение с дитем размещено успешно!" with title "Git clone"'
else
    osascript -e 'display alert "Ошибка при размещении"'
fi
